//Q. Check if given array is sorted or not
//solution:
#include <iostream>

using namespace std;
bool isSorted(int* arr, int n){
     if(n == 1 || n == 1){
         return true;
     }  
     if(arr[0]>arr[1]){
         return false;
     }
     return isSorted(arr+1,n-1);
}
int main()
{
    int n;
    cin>>n;
    int * arr = new int[n];
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    bool ans = isSorted(arr, n);
    switch(ans){
        case 0:
            cout<<"Not Sorted";
            break;
        case 1:
            cout<<"Sorted";
    }
}
