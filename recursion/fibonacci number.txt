Q.Find nth fibonacci number.
// fibonacci series 1 1 2 3 5 8 13 ... 
// 0th fibonacci = 1
// 1st fibonacci = 1
sol:
#include <iostream>

using namespace std;
int fibonacci(int n){
    if(n == 0 || n == 1){
        retu
    }
    return fibonacci(n-1) + fibonacci(n-2);
}
int main()
{
    int n;
    cin>>n;
    cout<<fibonacci(n)<<endl;
}
